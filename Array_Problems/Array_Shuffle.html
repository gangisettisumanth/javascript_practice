//18. **Array Shuffle**: Write a function to shuffle the elements of an array randomly. function shuffleArray(arr) { if (!Array.isArray(arr)) { return []; } const shuffledArray = [...arr]; for (let i = shuffledArray.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [shuffledArray[i], shuffledArray[j]] = [shuffledArray[j], shuffledArray[i]]; } return shuffledArray; } const inputArray = [1, 2, 3, 4, 5]; const shuffledArray = shuffleArray(inputArray); console.log(shuffledArray);
